      # 安装

      使用 npm 或 yarn 来安装 zykj-common

      # with npm
npm install zykj-common -S

# with yarn
yarn add zykj-common

      # 使用

      import Vue from 'vue';
import Zykj from 'zykj-common';
import 'zykj-common/lib/style/index.css';

Vue.use(Zykj);

      # 按需引入

      如果你只想用到其中某个组件，你可以使用以下的安装方式.首先你要安装 "babel-plugin-component"插件

        yarn install -D babel-plugin-component

      然后：

      import Vue from 'vue';
import {Example} from 'zykj-common';

Vue.use(Zykj);

// babel.config.js
 plugins: [
    [
      'component',
      {
        libraryName: 'zykj-common',
        styleLibraryName: 'style'
      },
      'zykj-common'
    ]
  ]
